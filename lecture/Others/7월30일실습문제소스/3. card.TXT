T = int(input())

for tc in range(1, T + 1):

    N = int(input())
    cards = input()
    cnt = [0] * 10

    for i in range(N):
        cnt[ int(cards[i]) ] += 1

    maxI = 0
    for i in range(10):
        if cnt[maxI] <= cnt[i]:
            maxI = i

    print("#%d %d %d" % (tc, maxI, cnt[maxI]))
