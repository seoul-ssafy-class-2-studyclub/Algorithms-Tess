import sys
sys.stdin = open("input.txt", "r")

def BruteForce(t, p) :
    i = 0
    j = 0
    N = len(t)
    M = len(p)
    while j < M and i < N :
        if t[i] != p[j] :
            i = i - j
            j = -1
        i += 1
        j += 1
    if j == M : return 1
    else: return 0


TC = int(input())

for tc in range(1, TC + 1):
    str1 = input()
    str2 = input()

    print("#%d"%tc, BruteForce(str2, str1))
